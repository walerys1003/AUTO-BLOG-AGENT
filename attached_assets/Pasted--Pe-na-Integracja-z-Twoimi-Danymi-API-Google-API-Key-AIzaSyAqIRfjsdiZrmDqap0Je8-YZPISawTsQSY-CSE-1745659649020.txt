import requests

def upload_image_to_wordpress(image_url, wp_url, wp_user, wp_pass):
    image_data = requests.get(image_url).content
    filename = image_url.split('/')[-1]
    media_endpoint = wp_url + '/wp-json/wp/v2/media'
    headers = {
        'Content-Disposition': f'attachment; filename={filename}',
    }

    response = requests.post(media_endpoint,
                             headers=headers,
                             data=image_data,
                             auth=(wp_user, wp_pass))

    if response.status_code == 201:
        print("Obrazek za≈Çadowany:", response.json()['source_url'])
        return response.json()['id'], response.json()['source_url']
    else:
        print("B≈ÇƒÖd:", response.content)
        return None, None
